/*====================================================================*/
/*==================== RAPPELS AVANT DE COMMENCER ====================*/
/*====================================================================*/

// line => ...
// CECI EST UNE FONCTION ANONYME.
// ICI, line REPRÉSENTE CHAQUE LIGNE LUE DANS LE FICHIER.

// word => ...
// ICI, word REPRÉSENTE CHAQUE MOT OBTENU APRÈS LE DÉCOUPAGE DES LIGNES.

// (word, 1)
// CECI CRÉE UNE PAIRE CLÉ-VALEUR.
// LA CLÉ EST LE MOT, ET LA VALEUR 1 REPRÉSENTE UNE OCCURRENCE DE CE MOT.

// _ + _
// CECI EST UNE ÉCRITURE ABRÉGÉE EN SCALA.
// ELLE SIGNIFIE : PRENDRE DEUX VALEURS ET LES ADDITIONNER.
// PAR EXEMPLE : (a, b) => a + b

// reduceByKey(_ + _)
// CETTE MÉTHODE REGROUPE LES VALEURS QUI PARTAGENT LA MÊME CLÉ,
// PUIS ELLE LES ADDITIONNE.
// EXEMPLE :
// ("Emma", 1), ("Emma", 1), ("Emma", 1) DEVIENT ("Emma", 3)

/*----------------------------------------------------------------------*/
/* ÉTAPE 1 - LECTURE DU FICHIER TEXTE                                   */
/*----------------------------------------------------------------------*/
scala> val inputTextRDD = sc.textFile("C:/Users/user1/Desktop/SPARK/etudiants.txt")
scala> inputTextRDD.collect()

/*----------------------------------------------------------------------*/
/* ÉTAPE 2 - DÉCOUPAGE DE CHAQUE LIGNE EN MOTS                          */
/*----------------------------------------------------------------------*/
scala> val tokenRDD = inputTextRDD.flatMap(line => line.split(" "))
scala> tokenRDD.collect()

/*----------------------------------------------------------------------*/
/* EXPLICATION : flatMap(line => line.split(" "))                       */
/*----------------------------------------------------------------------*/
// split(" ") DÉCOUPE CHAQUE LIGNE SELON LES ESPACES.
// flatMap APPLIQUE CE DÉCOUPAGE À CHAQUE LIGNE
// ET RETOURNE UNE SEULE SÉQUENCE PLATE DE MOTS.

/*----------------------------------------------------------------------*/
/* ÉTAPE 3 - TRANSFORMATION DE CHAQUE MOT EN PAIRE (mot, 1)             */
/*----------------------------------------------------------------------*/
scala> val wordCountPairRDD = tokenRDD.map(word => (word, 1))
scala> wordCountPairRDD.collect()

/*----------------------------------------------------------------------*/
/* EXPLICATION : map(word => (word, 1))                                 */
/*----------------------------------------------------------------------*/
// CHAQUE MOT EST TRANSFORMÉ EN PAIRE.
// EXEMPLE : "Emma" DEVIENT ("Emma", 1)
// CELA PRÉPARE LES DONNÉES POUR LE COMPTAGE PAR MOT.

/*----------------------------------------------------------------------*/
/* ÉTAPE 4 - AGRÉGATION DES OCCURRENCES PAR MOT                         */
/*----------------------------------------------------------------------*/
scala> val aggregatedWordCountRDD = wordCountPairRDD.reduceByKey(_ + _)
scala> aggregatedWordCountRDD.collect()

/*----------------------------------------------------------------------*/
/* EXPLICATION : reduceByKey(_ + _)                                     */
/*----------------------------------------------------------------------*/
// SPARK REGROUPE LES PAIRES AYANT LA MÊME CLÉ,
// PUIS ADDITIONNE LES VALEURS ASSOCIÉES.
// EXEMPLE :
// ("Adam", 1), ("Adam", 1) DEVIENT ("Adam", 2)

/*----------------------------------------------------------------------*/
/* ÉTAPE 5 - SAUVEGARDE DU RÉSULTAT DANS UN DOSSIER DE SORTIE           */
/*----------------------------------------------------------------------*/

/*----------------------------------------------------------------------*/

/*----------------------------------------------------------------------*/

/*----------------------------------------------------------------------*/
/* ÉTAPE 1 - CRÉATION D’UN RDD À PARTIR DE CHAÎNES "clé,valeur"         */
/*----------------------------------------------------------------------*/

/*----------------------------------------------------------------------*/
/* ÉTAPE 2 - TRANSFORMATION EN PAIRES (clé, valeur)                     */
/*----------------------------------------------------------------------*/
scala> val parsedKeyValueRDD = rawKeyValueRDD.map(record => (record.split(",")(0), record.split(",")(1).toInt))

/*----------------------------------------------------------------------*/
/* ÉTAPE 3 - REGROUPEMENT DES VALEURS PAR CLÉ AVEC groupByKey           */

/*----------------------------------------------------------------------*/
scala> parsedKeyValueRDD.reduceByKey((x, y) => x + y).collect()

/* ÉTAPE 5 - STOCKAGE DU RÉSULTAT DE reduceByKey                        */
/*----------------------------------------------------------------------*/
scala> val reducedByKeyRDD = parsedKeyValueRDD.reduceByKey((x, y) => x + y)

/*----------------------------------------------------------------------*/
/* EXPLICATION : groupByKey VS reduceByKey                              */
/*----------------------------------------------------------------------*/
// groupByKey REGROUPE TOUTES LES VALEURS D’UNE MÊME CLÉ.
// EXEMPLE :

//
// reduceByKey REGROUPE LES VALEURS D’UNE MÊME CLÉ,
// PUIS APPLIQUE UNE FONCTION D’AGRÉGATION.
// EXEMPLE :
// ("1", 14), ("1", 3) DEVIENT ("1", 17)
///
/// EN PRATIQUE, reduceByKey EST SOUVENT PRÉFÉRABLE POUR LES SOMMES,
// CAR IL ÉVITE DE CONSERVER TOUTES LES VALEURS SÉPARÉMENT
// QUAND ON VEUT UNIQUEMENT LE RÉSULTAT FINAL.

/*====================================================================*/
/*==================== DEUX POINTS IMPORTANTS POUR VOUS ==============*/
/*====================================================================*/

// PREMIER POINT IMPORTANT
// NOUS AVONS ÉCRIT sortBy(entry => entry._1)
// AU LIEU DE sortBy(a => a)
//
// POURQUOI ?
// PARCE QUE C’EST PLUS CLAIR À LIRE.
// ON VOIT IMMÉDIATEMENT QUE entry EST UNE PAIRE
// ET QUE entry._1 REPRÉSENTE LA CLÉ SUR LAQUELLE ON VEUT TRIER.
//
// DEUXIÈME POINT IMPORTANT
// CONCERNANT reduce ET fold,
// IL NE FAUT PAS DIRE QUE reduce RASSEMBLE TOUTES LES DONNÉES
// AVANT DE COMMENCER LE CALCUL.
//
// EN PRATIQUE, SPARK PEUT EFFECTUER DES CALCULS PARTIELS
// DANS CHAQUE PARTITION, PUIS COMBINER LES RÉSULTATS.
//
// LA DIFFÉRENCE MAJEURE EST QUE fold AJOUTE UNE VALEUR INITIALE,
// ALORS QUE reduce N’EN AJOUTE PAS.



/**********************************************************************/
/*********************** VOIR LES RÉPARTITIONS AVEC glom **************/
/**********************************************************************/

/*----------------------------------------------------------------------*/
/* EXEMPLE DE RDD RÉPARTI SUR 2 PARTITIONS                              */
/*----------------------------------------------------------------------*/
scala> val partitionedNumberRDD = sc.parallelize(1 to 7, 2)
scala> partitionedNumberRDD.collect()
scala> partitionedNumberRDD.glom().collect()

/*----------------------------------------------------------------------*/
/* EXPLICATION : glom()                                                 */
/*----------------------------------------------------------------------*/
// glom() PERMET DE VOIR LE CONTENU DE CHAQUE PARTITION SOUS FORME DE TABLEAU.
// C’EST TRÈS UTILE POUR COMPRENDRE COMMENT SPARK RÉPARTIT LES DONNÉES.
//
// EXEMPLE POSSIBLE :
// Array(Array(1, 2, 3), Array(4, 5, 6, 7))



/**********************************************************************/
/*********************** TROIS ACTIONS : reduce, fold, collect ********/
/**********************************************************************/

scala> val reduceExampleRDD = sc.parallelize(1 to 7, 2)
scala> reduceExampleRDD.collect()
scala> reduceExampleRDD.glom().collect()
scala> reduceExampleRDD.reduce((x, y) => x + y)

/*----------------------------------------------------------------------*/
/* EXPLICATION : reduce((x, y) => x + y)                                */
/*----------------------------------------------------------------------*/
// reduce COMBINE LES ÉLÉMENTS DU RDD DEUX À DEUX
// EN UTILISANT LA FONCTION FOURNIE.
//
// ICI :
// x + y SIGNIFIE QUE L’ON ADDITIONNE PROGRESSIVEMENT LES VALEURS.
//
// ILLUSTRATION SIMPLIFIÉE :
// 1 + 2 = 3
// 3 + 3 = 6
// 6 + 4 = 10
// 10 + 5 = 15
// 15 + 6 = 21
// 21 + 7 = 28
//
// RÉSULTAT FINAL : 28

/*----------------------------------------------------------------------*/
/* RAPPEL IMPORTANT                                                     */
/*----------------------------------------------------------------------*/
// reduce N’A PAS DE VALEUR INITIALE.
// IL UTILISE DIRECTEMENT LES ÉLÉMENTS DÉJÀ PRÉSENTS DANS LE RDD.
//
// DANS UN CONTEXTE DISTRIBUÉ, LE CALCUL PEUT ÊTRE FAIT PARTIELLEMENT
// DANS LES PARTITIONS, PUIS COMBINÉ ENSUITE.



/**********************************************************************/
/*********************** QUESTION SUR LE CALCUL DISTRIBUÉ *************/
/**********************************************************************/

// RAPPEL POUR LE FONCTIONNEMENT DU CALCUL SUR PLUSIEURS NŒUDS
// OU PLUSIEURS PARTITIONS :
// CONSULTER LA PREMIÈRE DIAPOSITIVE DU CHAPITRE D’INTRODUCTION À SPARK.
//
// L’IDÉE IMPORTANTE EST LA SUIVANTE :
// SPARK PEUT EFFECTUER DES CALCULS PARTIELS SUR CHAQUE PARTITION,
// PUIS COMBINER LES RÉSULTATS.



/**********************************************************************/
/******************************* FOLD *********************************/
/**********************************************************************/

/*----------------------------------------------------------------------*/
/* EXEMPLE DE fold AVEC VALEUR INITIALE                                 */
/*----------------------------------------------------------------------*/
scala> val foldExampleRDD = sc.parallelize(1 to 5, 2)
scala> foldExampleRDD.glom().collect()
scala> foldExampleRDD.fold(1)((x, y) => x + y)

/*----------------------------------------------------------------------*/
/* EXPLICATION : fold(1)((x, y) => x + y)                               */
/*----------------------------------------------------------------------*/
// fold RESSEMBLE À reduce,
// MAIS IL AJOUTE UNE VALEUR INITIALE, ICI : 1.
//
// CETTE VALEUR INITIALE EST UTILISÉE DANS LES CALCULS PARTIELS,
// PUIS LORS DE LA COMBINAISON FINALE.
//
// SI LE RDD EST RÉPARTI AINSI :
// Array(Array(1, 2), Array(3, 4, 5))
//
// ILLUSTRATION PÉDAGOGIQUE DU CALCUL DISTRIBUÉ :
// PARTITION 1 : 1 + 1 + 2 = 4
// PARTITION 2 : 1 + 3 + 4 + 5 = 13
// COMBINAISON FINALE : 1 + 4 + 13 = 18
//
// RÉSULTAT FINAL : 18

/*----------------------------------------------------------------------*/
/* REMARQUE PÉDAGOGIQUE IMPORTANTE                                      */
/*----------------------------------------------------------------------*/
// fold EST TRÈS UTILE POUR COMPRENDRE LE RÔLE DE LA VALEUR INITIALE
// DANS UN CALCUL DISTRIBUÉ.
//
// IL FAUT FAIRE ATTENTION :
// LA VALEUR INITIALE PEUT INTERVENIR PLUSIEURS FOIS DANS LE CALCUL,
// SELON LE NOMBRE DE PARTITIONS.

/*----------------------------------------------------------------------*/
/* FONCTION sample                                                      */
/*----------------------------------------------------------------------*/
scala> val sampleSourceRDD = sc.parallelize(1 to 100)
scala> sampleSourceRDD.sample(true, 0.2, 5).collect()
scala> sampleSourceRDD.count()

/*----------------------------------------------------------------------*/
/* EXPLICATION : sample(withReplacement, fraction, seed)                */
/*----------------------------------------------------------------------*/
// sample PERMET DE CONSTRUIRE UN ÉCHANTILLON À PARTIR D’UN RDD.
//
// PREMIER PARAMÈTRE : withReplacement
// true  : UN ÉLÉMENT PEUT ÊTRE TIRÉ PLUSIEURS FOIS
// false : UN ÉLÉMENT N’EST PAS RÉPÉTÉ DANS LE MÊME ÉCHANTILLON
//
// DEUXIÈME PARAMÈTRE : fraction
// IL REPRÉSENTE LA PROPORTION APPROXIMATIVE DE L’ÉCHANTILLON.
// PAR EXEMPLE, 0.2 SIGNIFIE ENVIRON 20 % DES DONNÉES.
//
// TROISIÈME PARAMÈTRE : seed
// IL SERT À REPRODUIRE LE MÊME ÉCHANTILLON SI L’ON VEUT
// RÉPÉTER LA MÊME EXPÉRIENCE DANS LES MÊMES CONDITIONS.
//
// count() PERMET ENSUITE DE COMPARER LA TAILLE DU RDD INITIAL

/*----------------------------------------------------------------------*/

/*====================================================================*/

scala> numberRDD1.intersection(numberRDD2).collect()

/*----------------------------------------------------------------------*/

/*----------------------------------------------------------------------*/

/*====================================================================*/
/*====================== TRANSFORMATION : subtract ===================*/
/*====================================================================*/

scala> numberRDD1.subtract(numberRDD2).collect()
scala> numberRDD2.subtract(numberRDD1).collect()

/*----------------------------------------------------------------------*/
/* EXPLICATION : subtract(...)                                          */
/*----------------------------------------------------------------------*/
// subtract RETOURNE LES ÉLÉMENTS DU PREMIER RDD
// QUI N’EXISTENT PAS DANS LE SECOND.
//
// IL EST IMPORTANT DE BIEN OBSERVER QUE :
// A.subtract(B) N’EST PAS LA MÊME CHOSE QUE B.subtract(A)

/*----------------------------------------------------------------------*/
/* EXPLICATION : cartesian(...)                                         */
/*----------------------------------------------------------------------*/
// cartesian CONSTRUIT LE PRODUIT CARTÉSIEN ENTRE DEUX RDD.
// CHAQUE ÉLÉMENT DU PREMIER RDD EST ASSOCIÉ À TOUS LES ÉLÉMENTS DU SECOND.
//
// SI LE PREMIER RDD CONTIENT 5 ÉLÉMENTS

/*----------------------------------------------------------------------*/
/* EXPLICATION : distinct()                                             */
/*----------------------------------------------------------------------*/
// distinct SUPPRIME LES DOUBLONS DANS LE RDD.



/*====================================================================*/
/*======================= TRANSFORMATION : cogroup ===================*/
/*====================================================================*/

scala> val leftPairRDD = sc.parallelize(Array(("A", "1"), ("B", "2"), ("C", "3"), ("D", "4")))
scala> val rightPairRDD = sc.parallelize(Array(("A", "a"), ("B", "b"), ("C", "c"), ("D", "d")))

scala> leftPairRDD.cogroup(rightPairRDD).collect()
scala> rightPairRDD.cogroup(leftPairRDD).collect()

scala> val partialRightPairRDD = sc.parallelize(Array(("A", "a"), ("C", "c")))
scala> leftPairRDD.cogroup(partialRightPairRDD).collect()

/*----------------------------------------------------------------------*/
/* EXPLICATION : cogroup(...)                                           */
/*----------------------------------------------------------------------*/
// cogroup REGROUPE, POUR CHAQUE CLÉ,
// LES VALEURS ISSUES DU PREMIER RDD ET DU SECOND RDD.
//
// LE RÉSULTAT EST UNE STRUCTURE DU TYPE :
// (clé, (valeurs_du_premier_RDD, valeurs_du_second_RDD))
//
// CETTE MÉTHODE RAPPELLE LE COMPORTEMENT D’UNE JOINTURE ÉTENDUE,
// CAR ELLE PERMET D’OBSERVER CE QUI EXISTE À GAUCHE, À DROITE,
// OU DES DEUX CÔTÉS.

/*----------------------------------------------------------------------*/
/* QUESTION DE RÉFLEXION : QUE SE PASSE-T-IL SI L’ON INVERSE CLÉS ET    */
/* VALEURS DANS L’EXEMPLE PRÉCÉDENT ?                                   */
/*----------------------------------------------------------------------*/
scala> val invertedLeftPairRDD = sc.parallelize(Array(("1", "A"), ("2", "B"), ("3", "C"), ("4", "D")))
scala> invertedLeftPairRDD.cogroup(rightPairRDD).collect()

// ICI, LES CLÉS NE CORRESPONDENT PLUS AUX CLÉS DE rightPairRDD.
// IL EST DONC NORMAL D’OBSERVER UN RÉSULTAT TRÈS DIFFÉRENT.



/*====================================================================*/
/*============= JOIN - rightOuterJoin - leftOuterJoin - fullOuterJoin =*/
/*====================================================================*/

scala> leftPairRDD.collect()
scala> rightPairRDD.collect()
scala> leftPairRDD.join(rightPairRDD).collect()

scala> leftPairRDD.collect()
scala> partialRightPairRDD.collect()
scala> leftPairRDD.join(partialRightPairRDD).collect()
scala> leftPairRDD.rightOuterJoin(partialRightPairRDD).collect()
scala> leftPairRDD.leftOuterJoin(partialRightPairRDD).collect()
scala> leftPairRDD.fullOuterJoin(partialRightPairRDD).collect()
scala> leftPairRDD.cogroup(partialRightPairRDD).collect()

/*----------------------------------------------------------------------*/
/* EXPLICATION : join(...)                                              */
/*----------------------------------------------------------------------*/
// join RETOURNE UNIQUEMENT LES CLÉS PRÉSENTES DANS LES DEUX RDD.
// C’EST L’ÉQUIVALENT D’UNE JOINTURE INTERNE.

/*----------------------------------------------------------------------*/
/* EXPLICATION : leftOuterJoin(...)                                     */
/*----------------------------------------------------------------------*/
// leftOuterJoin CONSERVE TOUTES LES CLÉS DU RDD DE GAUCHE,
// MÊME SI ELLES N’EXISTENT PAS DANS LE RDD DE DROITE.

/*----------------------------------------------------------------------*/
/* EXPLICATION : rightOuterJoin(...)                                    */
/*----------------------------------------------------------------------*/
// rightOuterJoin CONSERVE TOUTES LES CLÉS DU RDD DE DROITE,
// MÊME SI ELLES N’EXISTENT PAS DANS LE RDD DE GAUCHE.

/*----------------------------------------------------------------------*/
/* EXPLICATION : fullOuterJoin(...)                                     */
/*----------------------------------------------------------------------*/
// fullOuterJoin CONSERVE TOUTES LES CLÉS DES DEUX RDD,
// QU’ELLES SOIENT COMMUNES OU NON.

/*----------------------------------------------------------------------*/

/*----------------------------------------------------------------------*/
/* EXPLICATION : top(n)                                                 */
/*----------------------------------------------------------------------*/
// top(n) RETOURNE LES n PLUS GRANDES VALEURS.

/*====================================================================*/

/*====================================================================*/

/*----------------------------------------------------------------------*/
/* EXPLICATION : takeOrdered(n)                                         */
/*----------------------------------------------------------------------*/
// takeOrdered(n) RETOURNE LES n PLUS PETITES VALEURS,
// EN ORDRE CROISSANT.

// CONTRAIREMENT À top(n),
// QUI RETOURNE LES PLUS GRANDES VALEURS.



/*====================================================================*/
/*==================== COMPARAISON : top VS takeOrdered ==============*/
/*====================================================================*/

// top(n)         -> PLUS GRANDES VALEURS
// takeOrdered(n) -> PLUS PETITES VALEURS

// EXEMPLE :
// SUR LA LISTE List(1, 12, 5, -3, 4)
//
// top(2) PEUT RETOURNER : Array(12, 5)
// takeOrdered(2) RETOURNE : Array(-3, 1)

/*====================================================================*/

/*====================================================================*/

scala> val rawTextRDD = sc.parallelize(List("1 val1", "2 val2", "3 val3", "4 val4"))

/*----------------------------------------------------------------------*/

scala> rawTextRDD.keyBy(record => record.split(" ")(0)).collect()

/*----------------------------------------------------------------------*/
scala> rawTextRDD.map(record => (record.split(" ")(0), record.split(" ")(1))).collect()

/*----------------------------------------------------------------------*/
/* EXPLICATION : keyBy(...)                                             */
/*----------------------------------------------------------------------*/
// keyBy UTILISE UNE FONCTION POUR CALCULER LA CLÉ,
// MAIS IL CONSERVE LA VALEUR ORIGINALE COMME DEUXIÈME ÉLÉMENT DE LA PAIRE.
//
// EXEMPLE :
// "1 val1" DEVIENT ("1", "1 val1")
//
// IL EST DONC IMPORTANT DE COMPARER keyBy AVEC map,
// CAR map PERMET DE CHOISIR LIBREMENT LA STRUCTURE FINALE.

//
// EXEMPLE :

/*====================================================================*/

scala> import org.apache.spark.sql.SQLContext
scala> val sqlContext = new SQLContext(sc)

scala> val titanicDF = sqlContext
         .read
         .format("csv")
         .option("header", "true")
         .option("inferSchema", "true")
         .load("C:/Users/user1/Desktop/SPARK/titanic.csv")

scala> titanicDF.printSchema()
scala> titanicDF.select("name").show()

/*----------------------------------------------------------------------*/
/* QUESTION DE RÉFLEXION                                                */
/*----------------------------------------------------------------------*/
// COMBIEN DE FOIS UN CERTAIN NOM APPARAÎT-IL DANS LE TITANIC ?
// NOUS ALLONS UTILISER LES RDD POUR EFFECTUER CE TYPE DE TRAITEMENT.



/*====================================================================*/
/*==================== CONVERSION D’UN DATAFRAME EN RDD ==============*/
/*====================================================================*/

scala> val dataframeToRDD = titanicDF.rdd

/*----------------------------------------------------------------------*/
/* CONVERTIR UNE COLONNE EN RDD DE CHAÎNES                              */
/*----------------------------------------------------------------------*/
scala> val nameColumnRDD = titanicDF.select("name").map(row => row.toString()).rdd
scala> nameColumnRDD.collect()

/*----------------------------------------------------------------------*/
/* EXPLICATION                                                          */
/*----------------------------------------------------------------------*/
// titanicDF.rdd CONVERTIT LE DATAFRAME COMPLET EN RDD DE Row.
//
// titanicDF.select("name").map(...).rdd
// PERMET DE SE CONCENTRER UNIQUEMENT SUR LA COLONNE "name".
//
// row.toString() PRODUIT UNE REPRÉSENTATION TEXTUELLE DE CHAQUE LIGNE,
// MAIS CETTE REPRÉSENTATION CONTIENT DES CARACTÈRES SUPPLÉMENTAIRES
// QU’IL FAUT ENSUITE NETTOYER.



/*====================================================================*/
/*======================= NETTOYAGE DES DONNÉES ======================*/
/*====================================================================*/

scala> val cleanedNameRDD1 = nameColumnRDD
         .map(value => value.replace("[", ""))
         .map(value => value.replace("]", ""))

scala> val cleanedNameRDD2 = nameColumnRDD
         .map(value => value.replace("[", ""))
         .map(value => value.replace("]", ""))
         .map(value => value.replace("(", ""))
         .map(value => value.replace(")", ""))

scala> val cleanedNameRDD3 = cleanedNameRDD2
         .map(value => value.replace("\"", ""))

/*----------------------------------------------------------------------*/
/* EXPLICATION DU NETTOYAGE                                             */
/*----------------------------------------------------------------------*/
// CES ÉTAPES SERVENT À SUPPRIMER LES CARACTÈRES NON DÉSIRÉS
// INTRODUITS PAR LA CONVERSION EN CHAÎNE.
//
// PAR EXEMPLE, ON PEUT SUPPRIMER :
// [ ] ( ) "
//
// APRÈS NETTOYAGE, LE TEXTE EST PLUS FACILE À DÉCOUPER
// ET À TRAITER AVEC LES TRANSFORMATIONS RDD.



/*====================================================================*/
/*==================== DÉCOUPAGE ET WORD COUNT SUR LES NOMS ==========*/
/*====================================================================*/

scala> val flattenedNameRDD = cleanedNameRDD3.flatMap(_.split(","))
scala> flattenedNameRDD.collect()

scala> val mappedNameRDD = flattenedNameRDD.map(value => (value, 1))
scala> mappedNameRDD.collect()

scala> val nameCountRDD = mappedNameRDD.reduceByKey(_ + _)
scala> nameCountRDD.collect()

/*----------------------------------------------------------------------*/
/* EXPLICATION : PIPELINE map-reduce                                    */
/*----------------------------------------------------------------------*/
// 1) flatMap(_.split(",")) DÉCOUPE LES CHAÎNES
// 2) map(value => (value, 1)) CRÉE DES PAIRES (clé, 1)
// 3) reduceByKey(_ + _) CALCULE LE NOMBRE D’OCCURRENCES DE CHAQUE CLÉ
//
// C’EST UNE FORME CLASSIQUE DE WORD COUNT APPLIQUÉE ICI AUX NOMS.



/*====================================================================*/
/*================ CHAÎNAGE COMPLET AVEC SAUVEGARDE DU RÉSULTAT ======*/
/*====================================================================*/

scala> nameColumnRDD
         .map(value => value.replace("[", ""))
         .map(value => value.replace("]", ""))
         .map(value => value.replace("(", ""))
         .map(value => value.replace(")", ""))
         .flatMap(value => value.split(","))
         .map(value => (value, 1))
         .reduceByKey(_ + _)
         .saveAsTextFile("C:/Users/user1/Desktop/SPARK/jack")

/*----------------------------------------------------------------------*/
/* EXPLICATION                                                          */
/*----------------------------------------------------------------------*/
// CE CHAÎNAGE EFFECTUE TOUT LE TRAITEMENT EN UNE SEULE EXPRESSION.
//
// IL MONTRE COMMENT PASSER :
// DATAFRAME -> COLONNE -> RDD -> NETTOYAGE -> DÉCOUPAGE -> COMPTAGE -> SAUVEGARDE

         .map(value => value.replace("[", ""))
         .map(value => value.replace("]", ""))
         .map(value => value.replace("(", ""))
         .map(value => value.replace(")", ""))
         .map(value => value.replace("\"", ""))

/*----------------------------------------------------------------------*/

         .map(value => value.replace("[", ""))
         .map(value => value.replace("]", ""))
         .map(value => value.replace("(", ""))
         .map(value => value.replace(")", ""))
         .map(value => value.replace("\"", ""))
         .flatMap(value => value.split(","))
         .filter(value => value.contains("Emil"))

         .reduceByKey(_ + _)
         .count()

/* EXPLICATION                                                          */
/*----------------------------------------------------------------------*/

/*----------------------------------------------------------------------*/

/*----------------------------------------------------------------------*/

/* EXPLICATION                                                          */
/*----------------------------------------------------------------------*/

/*----------------------------------------------------------------------*/

/*----------------------------------------------------------------------*/
// ICI, SPARK ÉCOUTE UN PORT RÉSEAU.
// LES DONNÉES REÇUES SONT DU TEXTE.
// ENSUITE, SPARK EFFECTUE UN WORD COUNT EN CONTINU.

/*----------------------------------------------------------------------*/
/* TERMINAL 1 - CODE SPARK STREAMING                                    */
/*----------------------------------------------------------------------*/
scala> import org.apache.spark._
scala> import org.apache.spark.streaming._
scala> import org.apache.spark.streaming.StreamingContext._

scala> val ssc = new StreamingContext(sc, Seconds(3))

scala> val lines = ssc.socketTextStream("localhost", 9988)
scala> val words = lines.flatMap(line => line.split(" "))
scala> val pairs = words.map(word => (word, 1))
scala> val wordCounts = pairs.reduceByKey(_ + _)

scala> wordCounts.print()

scala> ssc.start()
scala> ssc.awaitTermination()

// ICI, SPARK SURVEILLE UN DOSSIER.

/*----------------------------------------------------------------------*/
/* CODE STREAMING SUR DOSSIER                                           */
/*----------------------------------------------------------------------*/
scala> import org.apache.spark._
scala> import org.apache.spark.streaming._
scala> import org.apache.spark.streaming.StreamingContext._

scala> val ssc = new StreamingContext(sc, Seconds(3))

scala> val fileStream = ssc.textFileStream("C:/Users/user1/Desktop/SPARK/file1")

scala> fileStream.foreachRDD(rdd => {
         val wordCountRDD = rdd
           .flatMap(line => line.split(" "))
           .map(word => (word, 1))
           .reduceByKey(_ + _)

         wordCountRDD.saveAsTextFile("C:/Users/user1/Desktop/SPARK/file2")
       })

scala> ssc.start()
scala> ssc.awaitTermination()

/*----------------------------------------------------------------------*/
/* EXPLICATION TRÈS DÉTAILLÉE                                           */
/*----------------------------------------------------------------------*/
// textFileStream("C:/Users/user1/Desktop/SPARK/file1")
// -> SPARK SURVEILLE LE DOSSIER file1.

//    IL EST LU DANS LE PROCHAIN MICRO-BATCH.
//
// foreachRDD(rdd => { ... })
// -> POUR CHAQUE MICRO-BATCH,
//    SPARK FOURNIT UN RDD QUE NOUS POUVONS TRAITER.
//
// AUTREMENT DIT,
// LE FLUX GLOBAL EST UN DStream,

//

// - flatMap(...) DÉCOUPE LES LIGNES

/*----------------------------------------------------------------------*/

/*----------------------------------------------------------------------*/

         if (!rdd.isEmpty()) {
           val wordCountRDD = rdd
             .flatMap(line => line.split(" "))
             .map(word => (word, 1))
             .reduceByKey(_ + _)

         }
       })

scala> ssc.start()

scala> import org.apache.spark.streaming._
scala> import org.apache.spark.streaming.StreamingContext._

         .setAppName("FileCount")
         .setMaster("local[2]")
         .set("spark.driver.allowMultipleContexts", "true")

scala> val ssc = new StreamingContext(sc, Seconds(1))

scala> val fileStream = ssc.textFileStream("/opt/spark/examples/src/main/resources/file1")

scala> fileStream.foreachRDD(rdd => {
         val wordCountRDD = rdd
           .flatMap(line => line.split(" "))
           .map(word => (word, 1))
           .reduceByKey(_ + _)

         wordCountRDD.saveAsTextFile("/opt/spark/examples/src/main/resources/file2")
       })

scala> ssc.start()
scala> ssc.awaitTermination()

// PAR EXEMPLE LES 9 DERNIÈRES SECONDES,
// TOUT EN RECALCULANT LE RÉSULTAT RÉGULIÈREMENT.

/*----------------------------------------------------------------------*/
/* C’EST QUOI UNE SLIDING WINDOW ?                                      */
/*----------------------------------------------------------------------*/
// UNE SLIDING WINDOW EST UNE FENÊTRE DE TEMPS
// QUI "GLISSE" AU FIL DU TEMPS.
//
// ELLE EST DÉFINIE PAR DEUX PARAMÈTRES :
//
// 1) LA TAILLE DE LA FENÊTRE
// -> COMBIEN DE TEMPS DE DONNÉES ON VEUT OBSERVER
//
// 2) LE PAS DE GLISSEMENT
// -> TOUS LES COMBIEN DE TEMPS ON RECALCULE LE RÉSULTAT

/*----------------------------------------------------------------------*/
/* EXEMPLE INTUITIF                                                     */
/*----------------------------------------------------------------------*/
// SI NOUS ÉCRIVONS :
/*
reduceByKeyAndWindow(..., Seconds(9), Seconds(3))
*/
//
// ALORS :
// - LA FENÊTRE CONTIENT LES 9 DERNIÈRES SECONDES DE DONNÉES
// - LE CALCUL EST REFAIT TOUTES LES 3 SECONDES
//
// DONC :
// À t = 9s   -> ON REGARDE LES DONNÉES DE 0 À 9
// À t = 12s  -> ON REGARDE LES DONNÉES DE 3 À 12
// À t = 15s  -> ON REGARDE LES DONNÉES DE 6 À 15
//
// LA FENÊTRE AVANCE,
// ELLE GLISSE DANS LE TEMPS.

/*----------------------------------------------------------------------*/
/* CODE                                                                 */
/*----------------------------------------------------------------------*/
scala> import org.apache.spark._
scala> import org.apache.spark.streaming._
scala> import org.apache.spark.streaming.StreamingContext._

scala> val conf = new SparkConf()
         .setMaster("local[2]")
         .setAppName("NetworkWordCount")
         .set("spark.driver.allowMultipleContexts", "true")

scala> val ssc = new StreamingContext(conf, Seconds(3))

scala> val messageStream = ssc.socketTextStream("localhost", 9988)

scala> val wordPairStream = messageStream
         .flatMap(line => line.split(" "))
         .map(word => (word, 1))

scala> val windowWordCountStream =
         wordPairStream.reduceByKeyAndWindow(
           (x: Int, y: Int) => x + y,
           Seconds(9),
           Seconds(3)
         )

scala> windowWordCountStream.print()

scala> ssc.start()
scala> ssc.awaitTermination()
